<script>

const videos = [

  {

    title: "Big Buck Bunny",

    video: "https://test-videos.co.uk/vids/bigbuckbunny/mp4/h264/720/Big_Buck_Bunny_720_10s_1MB.mp4",

    thumb: "https://peach.blender.org/wp-content/uploads/title_anouncement.jpg"

  },

  {

    title: "Sintel Trailer",

    video: "https://media.w3.org/2010/05/sintel/trailer.mp4",

    thumb: "https://i.ytimg.com/vi/eRsGyueVLvQ/maxresdefault.jpg"

  },

  {

    title: "Tears of Steel",

    video: "https://commondatastorage.googleapis.com/gtv-videos-bucket/sample/TearsOfSteel.mp4",

    thumb: "https://i.ytimg.com/vi/R6MlUcmOul8/maxresdefault.jpg"

  }

];

 

const player = document.getElementById("player");

const title = document.getElementById("title");

const grid = document.getElementById("grid");

 

let current = 0;

 

function loadGrid() {

  videos.forEach((v, i) => {

    const div = document.createElement("div");

    div.className = "card";

    div.innerHTML = `

      <img src="${v.thumb}">

      <p>${v.title}</p>

    `;

    div.onclick = () => play(i);

    grid.appendChild(div);

  });

}

 

function play(i) {

  current = i;

  player.src = videos[i].video;

  title.textContent = videos[i].title;

  player.play();

}

 

player.addEventListener("ended", () => {

  current = (current + 1) % videos.length;

  play(current);

});

 

loadGrid();

play(0);

</script>
